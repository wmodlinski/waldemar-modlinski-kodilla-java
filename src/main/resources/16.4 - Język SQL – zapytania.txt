SELECT I.SUMMARY, L.NAME
FROM issues I
JOIN issueslists L ON L.ID = I.ISSUESLISTS_ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM issues I
JOIN users U ON U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_TASK
FROM users U
JOIN issues I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
